void Del_complete(multimap<string, string>& events, string info1, string info2){
    int j = 0;
    auto range = events.equal_range(info1);
    for(auto i = range.first; i != range.second; ++i){
        if(i -> first == info1){
            events.erase(i);
           j+=1;
        }
    }
    cout << "Deleted " << j << " events" << endl;
}
